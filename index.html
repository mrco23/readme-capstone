SQL — Create Database & Tables

Copy & paste langsung ke MySQL client (atau gunakan tool migration). 

CREATE DATABASE IF NOT EXISTS capstone_db;
USE capstone_db;

-- students
CREATE TABLE students (
  id INT AUTO_INCREMENT PRIMARY KEY,
  email VARCHAR(120) NOT NULL,
  password VARCHAR(255) NOT NULL,
  photo_profile VARCHAR(255) NULL,
  name VARCHAR(100) NOT NULL,
  learning_style VARCHAR(50) NULL,
  program VARCHAR(120) NULL,
  level VARCHAR(50) NOT NULL,
  joined_since DATE NOT NULL
);


-- daily_insights
CREATE TABLE daily_insights (
  id INT AUTO_INCREMENT PRIMARY KEY,
  student_id INT,
  js_module_progress_delta INT,
  time_spent_today INT,
  daily_goal INT,
  learning_style VARCHAR(50),
  created_at DATE DEFAULT (CURRENT_DATE),
  FOREIGN KEY (student_id) REFERENCES students(id)
);

-- weekly_activity
CREATE TABLE weekly_activity (
  id INT AUTO_INCREMENT PRIMARY KEY,
  student_id INT,
  total_hours FLOAT,
  avg_hours_per_day FLOAT,
  days_met_goal INT,
  delta_vs_last_week VARCHAR(50),
  created_at DATE DEFAULT (CURRENT_DATE),
  FOREIGN KEY (student_id) REFERENCES students(id)
);

-- weekly_activity_daily
CREATE TABLE weekly_activity_daily (
  id INT AUTO_INCREMENT PRIMARY KEY,
  weekly_activity_id INT,
  day VARCHAR(10),
  hours FLOAT,
  FOREIGN KEY (weekly_activity_id) REFERENCES weekly_activity(id)
);

-- quiz_scores
CREATE TABLE quiz_scores (
  id INT AUTO_INCREMENT PRIMARY KEY,
  student_id INT,
  course_name VARCHAR(50),
  score INT,
  FOREIGN KEY (student_id) REFERENCES students(id)
);

-- recommendations
CREATE TABLE recommendations (
  id INT AUTO_INCREMENT PRIMARY KEY,
  student_id INT,
  title VARCHAR(150),
  category VARCHAR(50),
  description TEXT,
  progress_percent INT,
  action_label VARCHAR(50),
  FOREIGN KEY (student_id) REFERENCES students(id)
);

-- module_progress
CREATE TABLE module_progress (
  id INT AUTO_INCREMENT PRIMARY KEY,
  student_id INT,
  progress_percent INT,
  tasks_completed INT,
  tasks_total INT,
  FOREIGN KEY (student_id) REFERENCES students(id)
);

-- module_grades
CREATE TABLE module_grades (
  id INT AUTO_INCREMENT PRIMARY KEY,
  module_progress_id INT,
  quiz INT,
  assignment INT,
  final_test INT,
  FOREIGN KEY (module_progress_id) REFERENCES module_progress(id)
);

Relasi Antar Tabel

  1. students adalah tabel pusat. Semua tabel lain punya student_id yang menunjuk ke students.id.
  2. daily_insights: satu siswa dapat banyak catatan harian (One-to-Many).
  3. weekly_activity: satu siswa dapat banyak rekap mingguan (One-to-Many).
  4. weekly_activity_daily: setiap record weekly_activity punya detail 7 hari (One-to-Many).
  5. quiz_scores: menyimpan nilai per mata pelajaran (One-to-Many dari students).
  6. recommendations: daftar rekomendasi per siswa (One-to-Many).
  7. module_progress → module_grades: module_progress menyimpan progres modul, module_grades menyimpan nilai terkait (One-to-One atau One-to-Many sesuai kebutuhan).

Fungsi Masing-Masing Tabel

  • students: data utama siswa (profil).
  • daily_insights: ringkasan & metrik harian (waktu, progress delta, goal).
  • weekly_activity: rekap mingguan (total jam, rata-rata, delta minggu lalu).
  • weekly_activity_daily: detail jam tiap hari untuk grafik.
  • quiz_scores: nilai kuis per course.
  • recommendations: materi yang direkomendasikan untuk siswa.
  • module_progress: persentase & jumlah tugas modul.
  • module_grades: nilai kuis/assignment/final untuk modul.

Kenapa Memilih Tipe Data Itu?

VARCHAR

Untuk teks pendek seperti nama, email, title. Hemat tempat dan cepat.

TEXT

Untuk deskripsi panjang (rekomendation.description) karena bisa berisi paragraf.

INT

Untuk angka bulat: id, score, tasks, percent. Mudah untuk operasi matematika dan indexing.

FLOAT

Untuk nilai jam yang bisa desimal (mis. 1.5 jam).

DATE

Untuk menyimpan tanggal secara baku (YYYY-MM-DD).
